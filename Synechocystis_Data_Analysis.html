normality_test<-Growth_rates|>

The test does not suggest non-normality of the data, which speaks in favour of the reliability of the analysis above.

Since the tests do not suggest non-normality, nor a difference in mean OD\(_{720}\), we can test normality further by not grouping by treatment:

CASY-counter data

Assumptions: Cells are spherical, no contaminants not captured in the blank.

Counts measured by the CASY-counter are in a sample that is undiluted for the MC samples, 1:50 for the BBB samples and 1:2 for the ABB samples. 10 \(\mu\)L of this dilution is then added to 10ml of CASY buffer. Of this mixture, 200 \(\mu\)L is measured by the CASY-counter.

First compute the mean cell counts for each channel over the 3 replicates

MC_CASY_cellcount<-CASY|>
  filter(sample_type=="MC")|>
  group_by(channel)|>
  summarise(mean_count_ml=mean(1001*5*counts))|> # Multiply by 5 for counts per ml
  left_join(Design)|>
  mutate(channel=factor(channel)) # makes the plots prettier

BBB_cellcount<-CASY|>
  filter(sample_type == "BBB")|>
  group_by(channel)|>
  summarise(BBB_count_ml=mean(1001*5*50*counts))|>
  left_join(Design) # Multiply by 50 to account for dilution

ABB_BBB_cellcount<-CASY|>
  filter(sample_type=="ABB")|>
  group_by(channel)|>
  summarise(ABB_count_ml=mean(1001*5*2*counts))|> # Multiply by 2 to account for dilution
  left_join(BBB_cellcount)

qPCR Data:

The molecular weight of the DNA was computed using the tool at https://oligocalc.eu/, with the settings set to dsDNA, no primers (since the primers are in the mastermix) nor modifications at either ’ end (since those are also in the mastermix). We begin by computing the standard curves for each replicate, and the relevant metrics to determine which curve to use.

Computation

MolW_DNA<-670331.8 # Computed using oligonucleotide tool at https://oligocalc.eu/
Na=6.02214076*10^23 # Avogadro's number

# Compute the dna concentration in mg per uL
qPCR_standard<-qPCR|>
  filter(is.na(channel))|>
  mutate(dna_ng=0.04/(dilution_factor*5), # Factor of 4 due to the 1:5 mixture with the master mix
         dna_fragment=dna_ng/(10^9)/MolW_DNA*Na,
         replicate=factor(replicate))
  

# Add copy of all rows with replicate "all" for plotting purposes
qPCR_standard <- bind_rows(qPCR_standard, qPCR_standard |>
  mutate(replicate = "all",
         replicate = factor(replicate)))


qPCR_standard_metrics<-qPCR_standard|>
  group_by(replicate)|>
  summarise(slope = coef(lm(Ct ~ log10(dna_fragment)))[2],
            E= (10^(-1/slope)/2)*100,
            R2 = summary(lm(Ct ~ log(dna_fragment)))$r.squared)


optimal_slope=-1/log10(2)
optimal_E_L=90
optimal_E_H=110

min_R2=0.98

dna_regression<-lm(log10(dna_fragment)~Ct,qPCR_standard|>filter(replicate=="1"))

Computation

qPCR_DNA<-qPCR|>
  filter(sample_class=="sample")|>
  mutate(log_10_dna_frag=predict(dna_regression, newdata = data.frame(Ct = Ct)),
         dna_frag_ul=10^log_10_dna_frag*dilution_factor*5*2) #times 5 due to dilution in mastermix, times 2 due to ABB being in 1:2 solution 

Ploidy<-qPCR_DNA|>
  select(channel, dna_frag_ul, replicate )|>
  mutate(channel=factor(channel))|>
  left_join(Broken_cell_count)|>
  mutate(broken_cell_ul = Broken_cell_count_ml/1000)|>
  mutate(ploidy=dna_frag_ul/broken_cell_ul)

Ploidy_per_channel<-Ploidy|>
  group_by(channel,replicate)|>
  summarise(ploidy=mean(dna_frag_ul)/mean(broken_cell_ul))|>
  left_join(Broken_cell_count|>select(channel, phosphate_mM))

Ploidy_per_group<-Ploidy|>
  group_by(phosphate_mM)|>
  summarise(mean_ploidy=mean(ploidy),
            CI1=t.test(ploidy)$conf.int[1],
